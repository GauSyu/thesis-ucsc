% LTeX: enabled=false
% A predefined page
%%-----------------------------------------
%% Abstract page
%%-----------------------------------------
%
\makeatletter
\NewDocumentEnvironment{abstract}{}{
	\clearpage
	\thispagestyle{plain}
	\addchaptertocentry{}{Abstract}
	\begin{center}
		{\textbf{\Large Abstract}\par\bigskip
		\@title\par\bigskip
		by\par\bigskip
		\@author}
	\end{center}
	\ifdefined\@keywords
		\paragraph*{Key words:} {\@keywords}
	\fi
	\par
}{
	\cleardoublepage
}
\makeatother
%%-----------------------------------------
%% Dedication page
%%-----------------------------------------
%
\RenewDocumentEnvironment{dedication}{}{
	\clearpage
	\thispagestyle{empty}
	\addchaptertocentry{}{Dedication}
	\begin{center}
		\ \ \ \
		\vfill
}{
		\vfill
	\end{center}
	\cleardoublepage
}
%%-----------------------------------------
%% Acknowledgements page
%%-----------------------------------------
%
\NewDocumentEnvironment{acknowledgement}{}{
	\clearpage
	\thispagestyle{plain}
	\addchaptertocentry{}{Acknowledgments}
	\begin{center}
		\textbf{\Large Acknowledgements}
	\end{center}
	\par
}{
	\cleardoublepage
}
%
%%-----------------------------------------
%% PDF metadata
%%-----------------------------------------
\makeatletter
\hypersetup{%
	pdftitle={@title}, % title of PDF
	pdfauthor={@author}, % author of PDF
}

\ifdefined\@keywords
\hypersetup{pdfkeywords={@keywords}} % keywords
\fi
\makeatother
